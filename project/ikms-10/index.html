2020-12-20

const trayIcon = new Tray(path.join(__dirname, "../assets/IconTemplate.png"));

trayIcon.on("click", function () {
  if (mainWindow === null) {
    return;
  }
  if (mainWindow.isVisible()) {
    mainWindow.hide();
  } else {
    mainWindow.show();
  }
});
