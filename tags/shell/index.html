输出 Hello World test.sh #!/bin/bash echo "Hello World !" #! 告诉系统解释此脚本文件的 Shell 程序。 输出变量 test.sh #!/bin/bash message='Hello World' echo $message # output: Hello World 拼接变量 test.sh #!/bin/bash name='Peng' echo "Hello, $name" # output: Hello, Peng 运行Shell脚本 $ chmod +x ./test.sh #使脚本具有执行权限 $ ./test.sh #执行脚本 or $ sh test.sh #执行脚本 创建一个提交 git 记录的 shell 脚本 deploy.sh #!/bin/bash deploy_time=$(date +%Y-%m-%d\ %H:%M:%S) # 发布时间，格式如：2021-02-26 11:32:54 git pull # 拉取最新代码 git add . # 添加所有代码到代码库 git commit -m "deploy on $deploy_time" # 提交代码 git push # 推送代码到服务器

使用shell脚本创建文件

Shell

文件名使用每日日期 newfile.sh #!/bin/bash file_name=$(date +%Y.%m.%d) touch "${file_name}.md" 写入内容并创建文件 newfile.sh #!/bin/bash file_name=$(date +%Y.%m.%d) echo "hello world" > "${file_name}.md" 批量创建文件 newfiles.sh #!/bin/bash filenames=() filenames[0]='CSS实现弹幕效果.md' filenames[1]='CSS实现动画.md' filenames[1]='CSS盒子模型.md' for filename in ${filenames[@]} do touch ${filename} done
