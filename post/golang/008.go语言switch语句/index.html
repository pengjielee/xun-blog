2022-12-04
Go

问好

package main

import (
	"fmt"
)

func main() {
	var name string
	fmt.Printf("请输入姓名：")
	fmt.Scanln(&name)

	switch name {
	case "jim":
		fmt.Println("你好 " + name)
	case "tom":
		fmt.Println("hello " + name)
	case "mike":
		fmt.Println("hi " + name)
	case "a", "b", "c":
		fmt.Println("输入错误")
	default:
		fmt.Println("您好 " + name)
	}
}
