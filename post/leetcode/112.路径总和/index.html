112.路径总和

2021-04-20
leetcode

题目

给你二叉树的根节点 root 和一个表示目标和的整数 targetSum ，判断该树中是否存在 根节点到叶子节点 的路径，这条路径上所有节点值相加等于目标和 targetSum 。

叶子节点 是指没有子节点的节点。

示例 1：
           5
          / \
         4    8
        /     / \
       11    13  4
      /  \        \
     7    2        1
输入：root = [5,4,8,11,null,13,4,7,2,null,null,null,1], targetSum = 22 ，（5 - 4 - 11 - 2）
输出：true

示例 2：
   1
  / \
 2   3
输入：root = [1,2,3], targetSum = 5
输出：false

示例 3：
输入：root = [1,2], targetSum = 0
输出：false

提示：

  • 树中节点的数目在范围 [0, 5000] 内
  • -1000 <= Node.val <= 1000
  • -1000 <= targetSum <= 1000

JS实现

1、广度优先搜索

/**
 * Definition for a binary tree node.
 * function TreeNode(val) {
 *     this.val = val;
 *     this.left = this.right = null;
 * }
 */
/**
 * @param {TreeNode} root
 * @param {number} sum
 * @return {boolean}
 */
var hasPathSum = function (root, sum) {
  if (root === null) {
    return false;
  }
  // 存储节点的队列
  const queue_node = [];
  // 存储根节点到这些节点的路径和的队列
  const queue_value = [];

  // 添加根节点
  queue_node.push(root);
  // 添加根节点的值 
  queue_value.push(root.val);

  while (queue_node.length > 0) {
    const current = queue_node.shift();
    const value = queue_value.shift();

    // 判断是否是叶子节点
    if (current.left === null && current.right === null) {
      // 判断是否等于sum
      if (value === sum) {
        return true;
      }
      continue;
    }
    if (current.left != null) {
      // 入队左节点
      queue_node.push(current.left);
      // 入队根节点到左节点的路径和
      queue_value.push(current.left.val + value);
    }

    if (current.right != null) {
      // 入队右节点
      queue_node.push(current.right);
      // 入队根节点到右节点的路径和
      queue_value.push(current.right.val + value);
    }
  }
  return false;
};

2、递归

var hasPathSum = function (root, targetSum) {
  if (root === null) {
    return false;
  }
  if (root.left === null && root.right === null) {
    return targetSum === root.val;
  }
  return (
    hasPathSum(root.left, targetSum - root.val) ||
    hasPathSum(root.right, targetSum - root.val)
  );
};
