使用shell脚本创建文件

2021-01-15
Shell

文件名使用每日日期

newfile.sh

#!/bin/bash

file_name=$(date +%Y.%m.%d)

touch "${file_name}.md"

写入内容并创建文件

newfile.sh

#!/bin/bash

file_name=$(date +%Y.%m.%d)

echo "hello world" > "${file_name}.md"

批量创建文件

newfiles.sh

#!/bin/bash

filenames=()
filenames[0]='CSS实现弹幕效果.md'
filenames[1]='CSS实现动画.md'
filenames[1]='CSS盒子模型.md'

for filename in ${filenames[@]}
do 
  touch ${filename}
done
