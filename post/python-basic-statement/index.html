if语句

1、if

print('请输入用户名：')
name = input()
if name == 'peng':
  print('登录成功')

2、if…else

print('请输入登录用户名：')
name = input()
print('请输入登录密码：')
password = input()

if name == 'admin' and password == '123456':
  print('登录成功!')
else:
  print('用户名或密码错误!!!')

3、if…elif…else

for-in语句

# Prints out: P,y,t,h,o,n
for letter in 'Python':
  print(letter)

# Prints out: red,green,blue
for color in ['red','green','blue']:
  print(color)

# Prints out: 0:red,1:green,2:blue
colors = ['red','green','blue']
for index in range(len(colors)):
  print('%s:%s' % (index, colors[index]))

# Prints out: 1,2,3,4
for number in range(1,5):
  print(number)

# Prints out: 1,2,3,4 and then it prints "number value reached 4"
for number in range(1,5):
  print(number)
else:
  print("number value reached %d" %(number))

while语句

# Prints out: 0,1,2,3,4,5,6,7,8
count = 0
while (count < 9):
  print(count)
  count = count + 1

break语句

# Prints out: P,y,t
for letter in 'Python':     
  if letter == 'h':
    break
  print(letter)

# Prints out: 0,1,2,3,4
count = 0
while (count < 9):
  print(count)
  count = count + 1
  if(count == 5):
    break

continue语句

# Prints out: P,y,t,o,n
for letter in 'Python':     
  if letter == 'h':
    continue
  print(letter)

# Prints out: 1,2,3,4,6,7,8,9
count = 0
while (count < 9):
  count = count + 1
  if(count == 5):
    continue
  print(count)

pass语句

pass 是空语句，是为了保持程序结构的完整性。pass 不做任何事情，一般用做占位语句。

# Prints out: P,y,t,pass,h,o,n
for letter in 'Python':
  if letter == 'h':
    pass
    print('pass')
  print(letter)

1、猜数字游戏

import random

answer = random.randint(1, 100)
counter = 0

while True:
  counter += 1
  number = int(input('请输入: '))
  if number < answer:
    print('大一点')
  elif number > answer:
    print('小一点')
  else:
    print('恭喜你猜对了!')
    break

print('你总共猜了%d次' % counter)

if counter > 7:
  print('你的智商余额明显不足')

4、打印9x9乘法表

for i in range(1, 10):
  for j in range(1, i + 1):
    print('%d*%d=%d' % (j, i, i * j), end='\t')
  print()
