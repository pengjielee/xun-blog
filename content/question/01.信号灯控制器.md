## 题目 

信号灯控制器  
用 React 实现一个信号灯（交通灯）控制器，要求：  
1. 默认情况下，  
  1.1. 红灯亮20秒，并且最后5秒闪烁  
  1.2. 绿灯亮20秒，并且最后5秒闪烁  
  1.3. 黄灯亮10秒  
  1.4. 次序为 红-绿-黄-红-绿-黄  
2. 灯的个数、颜色、持续时间、闪烁时间、灯光次序都可配置，如：  
   lights=[{color: '#fff', duration: 10000, twinkleDuration: 5000}, ... ]

## 参考 

```javascript
import React from 'react';
import ReactDOM from 'react-dom';

let lights = [
  {
    color: 'red',
    duration: 20000,
    twinkleDuration: 5000,
    on: false,
    flash: false,
    countdown: 0,
  },
  {
    color: 'green',
    duration: 20000,
    twinkleDuration: 5000,
    on: false,
    flash: false,
    countdown: 0,
  },
  {
    color: 'yellow',
    duration: 10000,
    twinkleDuration: 0,
    on: false,
    flash: false,
    countdown: 0,
  },
];

const sleep = time => {
  return new Promise(resolve => {
    setTimeout(resolve, time);
  });
};

class TrafficLight extends React.Component {
  constructor(props) {
    super(props);
    this.state = {
      idx: 0,
    };
  }

  componentDidMount() {
    const start = async () => {
      for (let idx in lights) {
        await this.openLight(idx);
      }
      start();
    };
    start();
  }

  async openLight(idx) {
    idx = +idx;
    const { duration, twinkleDuration } = lights[idx];
    const normaltime = duration - twinkleDuration;
    lights = lights.map((item, index) => {
      if (index === idx) {
        item.on = true;
      } else {
        item.on = false;
      }
      return item;
    });
    this.setState({
      lights: lights,
    });
    this.countDownStart(idx, duration);
    await sleep(normaltime);

    // 是否有闪烁
    if (twinkleDuration > 0) {
      this.flashStart(idx);
      await sleep(twinkleDuration);
      this.flashClose();
      this.countDownStop();
    }
  }

  // 开启倒计时
  countDownStart(idx, duration){
    let countdown = duration / 1000;
    this.timer = setInterval(() => {
      --countdown;
      lights = lights.map((item, index) => {
        if (item.on && index === idx) {
          item.countdown = countdown;
        } 
        return item;
      });
      this.setState({
        lights: lights,
      });
    }, 1000)
  }

  // 停止倒计时
  countDownStop(){
    this.timer && clearInterval(this.timer);
  }

  // 开始闪烁
  flashStart(idx){
    lights = lights.map((item, index) => {
      if (item.on && index === idx) {
        item.flash = true;
      } else {
        item.flash = false;
      }
      return item;
    });
    this.setState({
      lights: lights,
    });
  }

  // 关闭闪烁
  flashClose(){
    lights = lights.map((item, index) => {
      item.flash = false;
      return item;
    });
    this.setState({
      lights: lights,
    });
  }

  render() {
    const lightList = lights.map(item => {
      const filter = item.on ? 'grayscale(0%)' : 'grayscale(80%)';
      return (
        <li key={item.color} className="light-item">
          <div
            className={['light', item.flash ? 'flash' : ''].join(' ')}
            style={{ background: item.color, filter: filter }}

          >{ item.countdown != 0 ? item.countdown : ''}</div>
        </li>
      );
    });

    return (
      <div className="container">
        <ul className="light-list">{lightList}</ul>
      </div>
    );
  }
}

ReactDOM.render(
  <React.StrictMode>
    <TrafficLight />
  </React.StrictMode>,
  document.getElementById('root'),
);
```
