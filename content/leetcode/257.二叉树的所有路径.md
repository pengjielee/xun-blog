## 题目

给定一个二叉树，返回所有从根节点到叶子节点的路径。

说明: 叶子节点是指没有子节点的节点。

示例:
```
输入:

   1
 /   \
2     3
 \
  5

输出: ["1->2->5", "1->3"]

解释: 所有根节点到叶子节点的路径为: 1->2->5, 1->3
```

## JS实现

1、深度优先搜索

```javascript
/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {string[]}
 */
var binaryTreePaths = function (root) {
  const result = [];

  if (root === null) {
    return result;
  }

  const dfs = (root, path) => {
    if (root) {
      path.push(root.val);
      if (root.left === null && root.right === null) {
        result.push(path.slice().join("->"));
      }
      dfs(root.left, path);
      dfs(root.right, path);
      path.pop();
    }
  };

  dfs(root, []);

  return result;
};
```

2、广度优先搜索

```javascript
var binaryTreePaths = function (root) {
  const result = [];
  const map = new Map();

  if (root === null) {
    return result;
  }

  const queue_node = [];
  queue_node.push(root);

  const getPath = (node) => {
    let temp = [];
    while (node != null) {
      temp.push(node.val);
      node = map.get(node);
    }
    temp = temp.reverse().join('->');
    result.push(temp.slice());
  };

  while (queue_node.length > 0) {
    const node = queue_node.shift();

    if (node.left === null && node.right === null) {
      getPath(node);
    } else {
      if (node.left != null) {
        map.set(node.left, node);
        queue_node.push(node.left);
      }
      if (node.right != null) {
        map.set(node.right, node);
        queue_node.push(node.right);
      }
    }
  }
  return result;
};
```

```javascript
var binaryTreePaths = function (root) {
  const result = [];

  if (root === null) {
    return result;
  }

  const queue_node = [];
  const queue_path = [];
  queue_node.push(root);
  queue_path.push(root.val.toString());

  while (queue_node.length > 0) {
    const node = queue_node.shift();
    const path = queue_path.shift();

    if (node.left === null && node.right === null) {
      result.push(path);
    }

    if (node.left != null) {
      queue_node.push(node.left);
      queue_path.push(path + "->" + node.left.val.toString());
    }
    if (node.right != null) {
      queue_node.push(node.right);
      queue_path.push(path + "->" + node.right.val.toString());
    }
  }
  return result;
};
```
