## JS实现

```javascript
//实现1
var reverse1 = (str) => {
  return str.split("").reverse().join("");
};

//实现2
var reverse2 = (str) => {
  let result = "";
  for (let char of str) {
    result = char + result;
  }
  return result;
};

//实现3
var reverse3 = (str) => {
  return str.split("").reduce((result, char) => char + result);
};

var str = 'hello';
console.log(reverse1(str)); //'olleh'
console.log(reverse2(str)); //'olleh'
console.log(reverse3(str)); //'olleh'
```

## Go实现

实现1:

```go
package main

import (
	"fmt"
)

func main() {
	str := "hello"
	result := Reverse(str)
	fmt.Println(result) //olleh
}

func Reverse(s string) string {
	a := []rune(s)
	for i, j := 0, len(a)-1; i < j; i, j = i+1, j-1 {
		a[i], a[j] = a[j], a[i]
	}
	return string(a)
}
```

实现2:

```go
package main

import (
	"fmt"
)

func main() {
	str := "hello"
	result := Reverse(str)
	fmt.Println(result) //olleh
}

func Reverse(s string) string {
	a := func(s string) *[]rune {
		var b []rune
		for _, k := range []rune(s) {
			defer func(v rune) {
				b = append(b, v)
			}(k)
		}
		return &b
	}(s)
	return string(*a)
}
```
