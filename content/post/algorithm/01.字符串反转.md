## 代码

```javascript
//实现1
var reverse1 = (str) => {
  return str.split("").reverse().join("");
};

//实现2
var reverse2 = (str) => {
  let result = "";
  for (let char of str) {
    result = char + result;
  }
  return result;
};

//实现3
var reverse3 = (str) => {
  return str.split("").reduce((result, char) => char + result);
};

var str = 'hello';
console.log(reverse1(str)); //'olleh'
console.log(reverse2(str)); //'olleh'
console.log(reverse3(str)); //'olleh'
```
