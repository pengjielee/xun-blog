## 1. 

```javascript
function fn2(){
  for(var i=0;i<4;i++){
    var timer=setInterval(function(i,timer){
      console.log(i);
      clearInterval(timer)
    },10,i,timer);
  }
}
fn2();
```

输出：0 1 2 3 3 3 3 重复

## 2. 

```javascript
function fn1(){
  for(var i=0;i<4;i++){
    var timer=setTimeout(function(i){
      console.log(i);
      clearTimeout(timer)
    },10,i);
  }
}
fn1();
```

输出：0 1 2 

## 3. 

```javascript
const a = [];
for (var i = 0; i < 10; i++) {
  a.push(function () {
    return console.log(i);
  });
}
a[0]();
a[1]();
```

输出：10 10 

## 4.
