如何判断左右小括号是否全部匹配。如 (( ))()((((()))))？

```javascript
var isValid = function (str) {
  var stack = [];
  var map = {
    "(": ")",
    "[": "]",
    "{": "}",
  };

  for (var char of str) {
    if (char in map) {
      stack.push(char);
    } else {
      if (!stack.length || char != map[stack.pop()]) {
        return false;
      }
    }
  }

  // 如果最后stack 里没有元素了， 就一定是匹配的
  return !stack.length;
};
```
