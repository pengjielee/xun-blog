## 实现1

```javascript
const reverse = (str) => {
  return str.split("").reverse().join("");
};
```

## 实现2

```javascript
const reverse = (str) => {
  let result = "";
  for (let char of str) {
    result = char + result;
  }
  return result;
};
```

## 实现3

```javascript
const reverse = (str) => {
  return str.split("").reduce((result, char) => char + result);
};
```
