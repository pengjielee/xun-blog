查找第一个等于给定值的元素（有重复元素）

const binarySearch = (arr, value) => {
  let low = 0,
    high = arr.length - 1;

  while (low <= high) {
    const mid = parseInt(low + (high - low) / 2);

    if (arr[mid] === value) {
      if (mid === 0 || arr[mid - 1] != value) {
        return mid;
      } else {
        high = mid - 1;
      }
    } else if (arr[mid] < value) {
      low = mid + 1;
    } else {
      high = mid - 1;
    }
  }

  return -1;
};

var nums = [3, 5, 6, 6, 6, 10];
console.log(binarySearch(nums, 6)); // 2
